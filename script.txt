1. INNER JOIN (lo m치s usado)

Saca solo lo que coincide en ambas tablas.

SELECT a.id, a.nombre, b.total
FROM tablaA a
INNER JOIN tablaB b ON a.id = b.id_tablaA;

2. LEFT JOIN (todos de la izquierda aunque no haya coincidencia)

SELECT a.id, a.nombre, b.total
FROM tablaA a
LEFT JOIN tablaB b ON a.id = b.id_tablaA;


---

3. RIGHT JOIN (todos de la derecha aunque no haya coincidencia)

SELECT a.id, a.nombre, b.total
FROM tablaA a
RIGHT JOIN tablaB b ON a.id = b.id_tablaA;

游늵 Rankings y Totales

4. Ranking por cantidad (ej: top clientes por compras)

SELECT c.nombre, COUNT(f.id) AS total_compras
FROM clientes c
INNER JOIN facturas f ON c.id = f.id_cliente
GROUP BY c.id
ORDER BY total_compras DESC
LIMIT 10;


---

5. Ranking por valor total (ej: top clientes por dinero gastado)

SELECT c.nombre, SUM(f.total) AS total_gastado
FROM clientes c
INNER JOIN facturas f ON c.id = f.id_cliente
GROUP BY c.id
ORDER BY total_gastado DESC
LIMIT 10;


---

6. Total general de una tabla (ej: total vendido)

SELECT SUM(total) AS total_vendido
FROM facturas;


---

7. Promedio, m치ximo y m칤nimo (estad칤sticas r치pidas)

SELECT 
    AVG(total) AS promedio,
    MAX(total) AS maximo,
    MIN(total) AS minimo

8. Ranking con JOIN y condici칩n extra (ej: top productos m치s vendidos)

SELECT p.nombre, SUM(d.cantidad) AS total_vendido
